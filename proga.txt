#include <iostream> 
#include <cmath> 
#include <string> 
using namespace std; 
 
 
void PerfectSquare(int n) 
{ 
    if (sqrt(n) * sqrt(n) == n) cout << " - полный квадрат"; 
} 
 
void SumCheck(int n, void(perfect)(int)) 
{ 
    for (int i = 1; i <= n; i++)  
    { 
        for (int j = 1; j <= i; j++)  
        { 
            for (int k = 1; k <= i; k++)  
            { 
                if (j * j + k * k == i) 
                { 
                    cout << i; perfect(i); 
                    cout << endl; 
                    goto again; 
                } 
            } 
        } 
    again:; 
    } 
} 
 
 
 
int main()

}
